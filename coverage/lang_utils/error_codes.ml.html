let try_find_explanation code =
  match List.find_opt (fun (c, _) -> String.equal c code) error_codes with
  | None ->
    List.find_opt (fun (c, _, _) -> String.equal c code) warning_codes
    |> Option.map (fun (c, e, _ ) -> (c, e))
  | o -> o
