\section{Typing}

\subsection*{Contexts}

$$
\begin{array}{@{}lrcl@{}}
\text{(type)} & \x{typ} & ::= &
  \dots ~|~
  \K{nat} ~|~
  \K{int} ~|~
  \K{float} ~|~
  \K{char} ~|~
  \K{text} ~|~
  \K{var} \x{typ} ~|~
  \K{label} \x{typ} \\
\text{(context)} & C & ::= &
  \x{id} = \x{typ-params} \x{typ} ~|~
  \x{id} <: \x{typ-params} \x{typ} ~|~
  \x{id} : \x{typ}
\end{array}
$$

\subsection*{Type Parameters \hfill $\boxed{C \vdash \x{typ-params} \dashv C'}$}
